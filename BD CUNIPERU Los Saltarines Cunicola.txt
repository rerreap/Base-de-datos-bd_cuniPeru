-- Tabla: empleados
CREATE TABLE empleados (
    id_empleado INT AUTO_INCREMENT PRIMARY KEY,
    primer_nombre VARCHAR(50) NOT NULL,
    segundo_nombre VARCHAR(50),
    primer_apellido VARCHAR(50) NOT NULL,
    segundo_apellido VARCHAR(50),
    email VARCHAR(100) NOT NULL UNIQUE,
    telefono VARCHAR(15),
    estado ENUM('activo', 'inactivo') DEFAULT 'activo',
    usuarioRegistro INT NOT NULL,
    fechaRegistro DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuarioModificacion INT,
    fechaModificacion DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (usuarioRegistro) REFERENCES usuarios(id),
    FOREIGN KEY (usuarioModificacion) REFERENCES usuarios(id)
);

-- Tabla: usuarios
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL, -- almacenar la contraseña en formato hash
    empleado_id INT NOT NULL,
    usuarioRegistro INT NOT NULL,
    fechaRegistro DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuarioModificacion INT,
    fechaModificacion DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (empleado_id) REFERENCES empleados(id_empleado),
    FOREIGN KEY (usuarioRegistro) REFERENCES usuarios(id),
    FOREIGN KEY (usuarioModificacion) REFERENCES usuarios(id)
);

-- Tabla: roles
CREATE TABLE roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    descripcion TEXT,
    usuarioRegistro INT NOT NULL,
    fechaRegistro DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuarioModificacion INT,
    fechaModificacion DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (usuarioRegistro) REFERENCES usuarios(id),
    FOREIGN KEY (usuarioModificacion) REFERENCES usuarios(id)
);

-- Tabla: permisos
CREATE TABLE permisos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    descripcion TEXT,
    usuarioRegistro INT NOT NULL,
    fechaRegistro DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuarioModificacion INT,
    fechaModificacion DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (usuarioRegistro) REFERENCES usuarios(id),
    FOREIGN KEY (usuarioModificacion) REFERENCES usuarios(id)
);

-- Tabla: roles_permisos
CREATE TABLE roles_permisos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    rol_id INT NOT NULL,
    permiso_id INT NOT NULL,
    fecha_asignacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuario_asignacion INT NOT NULL,
    FOREIGN KEY (rol_id) REFERENCES roles(id) ON DELETE CASCADE,
    FOREIGN KEY (permiso_id) REFERENCES permisos(id) ON DELETE CASCADE,
    FOREIGN KEY (usuario_asignacion) REFERENCES usuarios(id)
);

-- Tabla: usuarios_roles
CREATE TABLE usuarios_roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    rol_id INT NOT NULL,
    fecha_asignacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuario_asignacion INT NOT NULL,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE,
    FOREIGN KEY (rol_id) REFERENCES roles(id) ON DELETE CASCADE,
    FOREIGN KEY (usuario_asignacion) REFERENCES usuarios(id)
);

-- Tabla: transacciones
CREATE TABLE transacciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    tipo_transaccion ENUM('crear', 'modificar', 'eliminar', 'login', 'logout') NOT NULL,
    descripcion TEXT,
    fecha_transaccion DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
);

-- Procedimiento para registrar la entrada de productos al almacén
DELIMITER //
CREATE PROCEDURE RegistrarEntrada(
    IN p_idProductoAlmacen INT,
    IN p_cantidad INT,
    IN p_usuarioModificacion INT
)
BEGIN
    UPDATE ProductosAlmacen
    SET cantidad = cantidad + p_cantidad,
        fechaModificacion = CURRENT_TIMESTAMP,
        usuarioModificacion = p_usuarioModificacion
    WHERE idProductoAlmacen = p_idProductoAlmacen;
END //
DELIMITER ;

-- Procedimiento para registrar la salida de productos del almacén
DELIMITER //
CREATE PROCEDURE RegistrarSalida(
    IN p_idProductoAlmacen INT,
    IN p_cantidad INT,
    IN p_usuarioModificacion INT
)
BEGIN
    DECLARE stockActual INT;

    -- Obtener la cantidad actual en stock
    SELECT cantidad INTO stockActual
    FROM ProductosAlmacen
    WHERE idProductoAlmacen = p_idProductoAlmacen;

    -- Verificar si hay suficiente stock
    IF stockActual >= p_cantidad THEN
        UPDATE ProductosAlmacen
        SET cantidad = cantidad - p_cantidad,
            fechaModificacion = CURRENT_TIMESTAMP,
            usuarioModificacion = p_usuarioModificacion
        WHERE idProductoAlmacen = p_idProductoAlmacen;
    ELSE
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'No hay suficiente stock para realizar esta salida.';
    END IF;
END //
DELIMITER ;

-- Procedimiento para registrar un pago en CuentasPorCobrar
DELIMITER //

CREATE PROCEDURE RegistrarPagoCobro(
    IN p_idCuentaCobrar INT,
    IN p_monto DECIMAL(10, 2),
    IN p_usuario INT
)
BEGIN
    -- Verificar si el usuario tiene permisos para registrar pagos
    DECLARE v_usuarioRol INT;
    DECLARE v_monto DECIMAL(10, 2);
    
    SELECT idRol INTO v_usuarioRol FROM Usuarios WHERE idUsuario = p_usuario; -- Asegúrate de tener la tabla Usuarios
    
    IF v_usuarioRol NOT IN (SELECT idRol FROM Roles WHERE nombre IN ('Administrador', 'Tesoreria')) THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'No tiene permiso para registrar pagos';
    ELSE
        SELECT monto INTO v_monto FROM CuentasPorCobrar WHERE idCuentaCobrar = p_idCuentaCobrar;
        
        IF v_monto >= p_monto THEN
            UPDATE CuentasPorCobrar
            SET monto = monto - p_monto, 
                estado = CASE WHEN monto - p_monto = 0 THEN 'Pagada' ELSE 'Pendiente' END
            WHERE idCuentaCobrar = p_idCuentaCobrar;
        ELSE
            SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'El monto a pagar supera el saldo pendiente';
        END IF;
    END IF;
END;
//

DELIMITER ;

-- CREACION DE PROCEDIMIENTO ALMACENADOS

-- Genera la edad del conejo

drop procedure if exists edad_actual;

delimiter //
create procedure edad_actual()
begin
select  Fecha_nacimiento, 
   YEAR( CURDATE( ) ) - YEAR( Fecha_nacimiento )  - 
		IF( MONTH( CURDATE( ) ) < MONTH( Fecha_nacimiento), 1,
		IF ( MONTH(CURDATE( )) = MONTH(Fecha_nacimiento),
		IF (DAY( CURDATE( ) ) < DAY( Fecha_nacimiento ),1,0 ),0)) AS años,

	MONTH(CURDATE()) - MONTH( Fecha_nacimiento) + 12 *
		IF( MONTH(CURDATE())<MONTH(Fecha_nacimiento), 1,
		IF(MONTH(CURDATE())=MONTH(Fecha_nacimiento),
		IF (DAY(CURDATE())<DAY(Fecha_nacimiento),1,0),0)) - 
			IF(MONTH(CURDATE())<>MONTH(Fecha_nacimiento),(DAY(CURDATE())<DAY(Fecha_nacimiento)), 
            IF (DAY(CURDATE())<DAY(Fecha_nacimiento),1,0 ) ) AS meses, 
	(DAY( CURDATE( ) ) - DAY( Fecha_nacimiento ) +30 * ( DAY(CURDATE( )) < DAY(Fecha_nacimiento) )) AS dias  
from registro_macho;
end 
// delimiter ;



-- Calcula dias para la palpacion
drop procedure if exists fecha_palpacion;

delimiter //
create procedure fecha_palpacion()
begin
  select fecha_monta, DATE_ADD(fecha_monta, INTERVAL 14 DAY) fec_palpacion from monta;
end 
// delimiter ;


-- Calcula dias para poner nido
drop procedure if exists fecha_nido;

delimiter //
create procedure fecha_nido()
begin
  select fecha_monta, DATE_ADD(fecha_monta, INTERVAL 29 DAY) feec_nido from monta;
end 
// delimiter ;


-- Calcula dias para parto
drop procedure if exists fecha_parto;

delimiter //
create procedure fecha_parto()
begin
  select fecha_monta, DATE_ADD(fecha_monta, INTERVAL 31 DAY) fec_probable_parto from monta;
end 
// delimiter ;
